## [41. First Missing Positive](https://leetcode.com/problems/first-missing-positive/description/)
```
/**
 * @param {number[]} nums
 * @return {number}
 */
var firstMissingPositive = function(nums) {
    var exp = 1;
    nums.sort((a,b) => a-b).forEach(num => {
        if (num > 0 && num === exp)  exp++;
    });
    return exp;
};
```
